[1720. 解码异或后的数组](https://leetcode-cn.com/problems/decode-xored-array/)

- A XOR B = C ==> A XOR C = B, B XOR C = A

```c++
class Solution {
public:
    vector<int> decode(vector<int>& encoded, int first) {
      
      vector<int> arr;
      if (encoded.size() == 0)
        return arr;
  
      arr.push_back(first);
      for (int i = 1; i <= encoded.size(); ++i) {
        arr.push_back(arr[i - 1] ^ encoded[i - 1]);
      }

      return arr;
    }
};
```



[1684. 统计一致字符串的数目](https://leetcode-cn.com/problems/count-the-number-of-consistent-strings/)

- (1) Set
- (2) Hash 26 个字母
- (3) 移位的思想

```C++
class Solution {
public:
    int countConsistentStrings(string allowed, vector<string>& words) {
      int mask = decode(allowed);
      int k = 0;
      for (int i = 0; i < words.size(); ++i) {
        k += (mask | decode(words[i])) == mask ? 1 : 0;
      }
      return k;
    }

 private:
  int decode(string word) {
    int code = 0;
    for (int i = 0; i < word.size(); ++i) {
      code |= (1 << (word[i] - 'a'));
    }
    return code;
  }
};
```



[1704. 判断字符串的两半是否相似](https://leetcode-cn.com/problems/determine-if-string-halves-are-alike/)

```c++
#include <unordered_map>
#include <iostream>

using namespace std;

class Solution {
public:
    bool halvesAreAlike(string s) {
      unordered_map<int, int> a, b;

      char list[10] = {'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'};
      
      int k = s.size() / 2;
      for (int i = 0; i < k; ++i)
        a[s[i] - 'A']++;
      for (int j = k; j < s.size(); ++j)
        b[s[j] - 'A']++;
      
      int count = 0;
      for (int i = 0; i < 10; ++i) {
        count += a[list[i] - 'A'];
        count -= b[list[i] - 'A'];
      }
      return count == 0;
    }
};
```
