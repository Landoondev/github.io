2、找最大和最小元素（用分治写）

```C++
static int lmax = 0, lmin = 0, rmax = 0, rmin = 0;
void max_min(int A[], int l, int r, int& fmax, int& fmin) {
  if (l > r) return;

  else if (l == r) {
    fmax = A[l];
    fmin = A[l];
  }
  else if (l == r - 1) {
    fmax = max(A[l], A[r]);
    fmin = min(A[l], A[r]);
  }
  else {
    // 分治
    int mid = l + (r - l) / 2;
    max_min(A, l, mid, lmax, lmin);
    max_min(A, mid+1, r, rmax, rmin);
    // 合并
    fmax = max(lmax, rmax);
    fmin = min(lmin, rmin);
  }
}

void max_min(int A[], int n, int& fmax, int& fmin) {
  max_min(A, 0, n-1, fmax, fmin);
}
```
